3// mDNS/UPnP service discovery using modern service discovery protocols

11#[derive(Debug, Clone, Serialize, Deserialize)]
12pub struct ServerInfo {
13    pub hostname: String,
14    pub ip: String,
15    pub port: u16,
16    pub version: String,
17    pub capabilities: Vec<String>,
18}
19
20pub struct DiscoveryService {
21    server_info: ServerInfo,
22    #[cfg(feature = "discovery")]
23    mdns_daemon: Option<mdns_sd::ServiceDaemon>,
24    advertising: bool,
25}

27impl DiscoveryService {
28    pub fn new(ip: String, port: u16) -> Result<Self> {
29        let server_info = ServerInfo {
30            hostname: hostname::get()
31                .map(|h| h.to_string_lossy().to_string())
32                .unwrap_or_else(|_| "dpstream-server".to_string()),
33            ip,
34            port,
35            version: env!("CARGO_PKG_VERSION").to_string(),
36            capabilities: vec![
37                "dolphin".to_string(),
38                "gamecube".to_string(),
39                "wii".to_string(),
40                "h264".to_string(),
41                "nvenc".to_string(),
42                "tailscale".to_string(),
43            ],
44        };
45
46        info!(
47            "Initializing discovery service for {}",
48            server_info.hostname
49        );
50        debug!("Server capabilities: {:?}", server_info.capabilities);
51
52        Ok(Self {
53            server_info,
54            #[cfg(feature = "discovery")]
55            mdns_daemon: None,
56            advertising: false,
57        })
58    }

60    pub async fn start_advertising(&mut self) -> Result<()> {
61        if self.advertising {
62            warn!("Discovery service already advertising");
63            return Ok(());
64        }
65
66        info!("Starting service discovery advertising");
67        info!("Service: {}:{}", self.server_info.ip, self.server_info.port);
68
69        #[cfg(feature = "discovery")]
70        {
71            self.start_mdns_advertising().await?;
72        }
73
74        #[cfg(not(feature = "discovery"))]
75        {
76            info!("mDNS discovery feature not enabled, using simulation mode");
77            self.simulate_advertising().await?;
78        }
79
80        self.advertising = true;
81        info!("Service discovery advertising started successfully");
82        Ok(())
83    }

85    #[cfg(feature = "discovery")]
86    async fn start_mdns_advertising(&mut self) -> Result<()> {
87        use mdns_sd::{ServiceDaemon, ServiceInfo};
88
89        let daemon = ServiceDaemon::new()
90            .map_err(|e| NetworkError::Discovery(format!("Failed to create mDNS daemon: {e}")))?;
91
92        let service_type = "_nvstream._tcp.local.";
93        let instance_name = format!("{}._nvstream._tcp.local.", self.server_info.hostname);
94
95        let properties = vec![
96            ("hostname".to_string(), self.server_info.hostname.clone()),
97            ("version".to_string(), self.server_info.version.clone()),
98            (
99                "capabilities".to_string(),
100                self.server_info.capabilities.join(","),
101            ),
102            ("GfeVersion".to_string(), "3.20.4.14".to_string()), // GameStream compatibility
103            ("mac".to_string(), "00:11:22:33:44:55".to_string()), // Placeholder MAC
104        ];
105
106        let service_info = ServiceInfo::new(
107            service_type,
108            &instance_name,
109            &self.server_info.hostname,
110            &self.server_info.ip,
111            self.server_info.port,
112            &properties[..],
113        )
114        .map_err(|e| NetworkError::Discovery(format!("Failed to create service info: {e}")))?;
115
116        daemon
117            .register(service_info)
118            .map_err(|e| NetworkError::Discovery(format!("Failed to register service: {e}")))?;
119
120        self.mdns_daemon = Some(daemon);
121        info!("mDNS service registered: {}", instance_name);
122        Ok(())
123    }

151    pub async fn stop_advertising(&mut self) -> Result<()> {
152        if !self.advertising {
153            debug!("Discovery service not currently advertising");
154            return Ok(());
155        }
156
157        info!("Stopping service discovery advertising");
158
159        #[cfg(feature = "discovery")]
160        {
161            if let Some(daemon) = self.mdns_daemon.take() {
162                // The daemon will automatically unregister services when dropped
163                drop(daemon);
164                info!("mDNS service unregistered");
165            }
166        }
167
168        self.advertising = false;
169        info!("Service discovery advertising stopped");
170        Ok(())
171    }

181    pub async fn discover_servers(timeout: Duration) -> Result<Vec<ServerInfo>> {
182        info!(
183            "Discovering dpstream servers on network (timeout: {:?})",
184            timeout
185        );
186
187        #[cfg(feature = "discovery")]
188        {
189            Self::discover_mdns_servers(timeout).await
190        }
191
192        #[cfg(not(feature = "discovery"))]
193        {
194            warn!("Discovery feature not enabled, returning empty server list");
195            Ok(vec![])
196        }
197    }

199    #[cfg(feature = "discovery")]
200    async fn discover_mdns_servers(timeout: Duration) -> Result<Vec<ServerInfo>> {
201        use mdns_sd::{ServiceDaemon, ServiceEvent};
202        use std::collections::HashMap;
203        use tokio::time::timeout as async_timeout;
204
205        let daemon = ServiceDaemon::new().map_err(|e| {
206            NetworkError::Discovery(format!("Failed to create discovery daemon: {e}"))
207        })?;
208
209        let service_type = "_nvstream._tcp.local.";
210        let receiver = daemon.browse(service_type).map_err(|e| {
211            NetworkError::Discovery(format!("Failed to start service browsing: {e}"))
212        })?;
213
214        let mut servers = HashMap::new(); // Use HashMap to deduplicate
215        let mut discovered_count = 0;
216
217        info!("Browsing for {} services...", service_type);
218
219        match async_timeout(timeout, async {
220            while let Ok(event) = receiver.recv_async().await {
221                match event {
222                    ServiceEvent::ServiceResolved(service) => {
223                        discovered_count += 1;
224                        debug!(
225                            "Discovered service #{}: {}",
226                            discovered_count,
227                            service.get_fullname()
228                        );
229
230                        let properties = service.get_properties();
231                        let hostname = properties
232                            .get("hostname")
233                            .map(|p| p.val_str().to_string())
234                            .unwrap_or_else(|| service.get_hostname().to_string());
235
236                        // Get all IP addresses and prefer IPv4
237                        let addresses: Vec<_> = service.get_addresses().iter().collect();
238                        let ip = addresses
239                            .iter()
240                            .find(|addr| addr.is_ipv4())
241                            .or_else(|| addresses.first())
242                            .map(|addr| addr.to_string())
243                            .unwrap_or_else(|| "unknown".to_string());
244
245                        let server_info = ServerInfo {
246                            hostname: hostname.clone(),
247                            ip,
248                            port: service.get_port(),
249                            version: properties
250                                .get("version")
251                                .map(|p| p.val_str().to_string())
252                                .unwrap_or_else(|| "unknown".to_string()),
253                            capabilities: properties
254                                .get("capabilities")
255                                .map(|caps| {
256                                    caps.val_str()
257                                        .split(',')
258                                        .map(|s| s.trim().to_string())
259                                        .filter(|s| !s.is_empty())
260                                        .collect()
261                                })
262                                .unwrap_or_default(),
263                        };
264
265                        info!(
266                            "Found server: {} at {}:{} (capabilities: {:?})",
267                            server_info.hostname,
268                            server_info.ip,
269                            server_info.port,
270                            server_info.capabilities
271                        );
272
273                        // Use hostname as key to deduplicate
274                        servers.insert(hostname, server_info);
275
276                        // Early exit if we found enough servers
277                        if servers.len() >= 10 {
278                            break;
279                        }
280                    }
281                    ServiceEvent::ServiceRemoved(_, fullname) => {
282                        debug!("Service removed from network: {}", fullname);
283                        // Remove from our list if it exists
284                        servers.retain(|_, server| {
285                            let service_name = format!("{}._nvstream._tcp.local.", server.hostname);
286                            service_name != fullname
287                        });
288                    }
289                    ServiceEvent::SearchStarted(_) => {
290                        debug!("mDNS search started");
291                    }
292                    ServiceEvent::SearchStopped(_) => {
293                        debug!("mDNS search stopped");
294                        break;
295                    }
296                    _ => {
297                        debug!("Other mDNS event received");
298                    }
299                }
300            }
301        })
302        .await
303        {
304            Ok(_) => {
305                debug!("Discovery completed normally");
306            }
307            Err(_) => {
308                debug!("Discovery timeout reached after {:?}", timeout);
309            }
310        }
311
312        let server_list: Vec<ServerInfo> = servers.into_values().collect();
313        info!(
314            "Discovery completed, found {} unique servers from {} total discoveries",
315            server_list.len(),
316            discovered_count
317        );
318
319        Ok(server_list)
320    }
321}
