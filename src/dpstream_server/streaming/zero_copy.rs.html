15/// Zero-copy video buffer with pre-allocated memory pools
16#[repr(align(64))] // Cache line aligned for optimal performance
17#[derive(Debug)]
18#[allow(dead_code)]
19pub struct ZeroCopyVideoBuffer {
20    /// Pre-allocated video data aligned for DMA operations
21    data: *mut u8,
22    /// Buffer capacity in bytes
23    capacity: usize,
24    /// Current data length
25    length: AtomicUsize,
26    /// Buffer ID for tracking and debugging
27    buffer_id: u64,
28    /// Reference count for safe sharing
29    ref_count: AtomicUsize,
30    /// Memory pool this buffer belongs to
31    pool_id: u32,
32}
33
34unsafe impl Send for ZeroCopyVideoBuffer {}
35unsafe impl Sync for ZeroCopyVideoBuffer {}
36
37#[allow(dead_code)]
38impl ZeroCopyVideoBuffer {
39    /// Create a new zero-copy buffer with specified capacity
40    pub fn new(capacity: usize, buffer_id: u64, pool_id: u32) -> Result<Self, PoolError> {
41        // Allocate aligned memory for optimal cache performance
42        let layout = std::alloc::Layout::from_size_align(capacity, 64)
43            .map_err(|_| PoolError::AllocationFailed)?;
44
45        let data = unsafe { std::alloc::alloc_zeroed(layout) };
46        if data.is_null() {
47            return Err(PoolError::AllocationFailed);
48        }
49
50        Ok(Self {
51            data,
52            capacity,
53            length: AtomicUsize::new(0),
54            buffer_id,
55            ref_count: AtomicUsize::new(1),
56            pool_id,
57        })
58    }
59
60    /// Get a slice of the current valid data
61    #[inline(always)]
62    pub fn data(&self) -> &[u8] {
63        let length = self.length.load(Ordering::Acquire);
64        unsafe { std::slice::from_raw_parts(self.data, length) }
65    }
66
67    /// Get a mutable slice for writing data (zero-copy)
68    #[inline(always)]
69    pub fn data_mut(&mut self) -> &mut [u8] {
70        unsafe { std::slice::from_raw_parts_mut(self.data, self.capacity) }
71    }
72
73    /// Set the valid data length
74    #[inline(always)]
75    pub fn set_length(&self, length: usize) {
76        debug_assert!(length <= self.capacity);
77        self.length.store(length, Ordering::Release);
78    }
79
80    /// Get buffer capacity
81    #[inline(always)]
82    pub fn capacity(&self) -> usize {
83        self.capacity
84    }
85
86    /// Get buffer ID for tracking
87    #[inline(always)]
88    pub fn id(&self) -> u64 {
89        self.buffer_id
90    }
91
92    /// Increment reference count
93    #[inline(always)]
94    pub fn add_ref(&self) {
95        self.ref_count.fetch_add(1, Ordering::Relaxed);
96    }
97
98    /// Decrement reference count and return true if should be freed
99    #[inline(always)]
100    pub fn release(&self) -> bool {
101        self.ref_count.fetch_sub(1, Ordering::Relaxed) == 1
102    }
103}

105impl Drop for ZeroCopyVideoBuffer {
106    fn drop(&mut self) {
107        unsafe {
108            let layout = std::alloc::Layout::from_size_align_unchecked(self.capacity, 64);
109            std::alloc::dealloc(self.data, layout);
110        }
111    }
112}

114/// High-performance video buffer pool with intelligent allocation strategies
115#[allow(dead_code)]
116pub struct VideoBufferPool {
117    /// Pre-allocated buffers for different resolutions
118    pools: RwLock<SmallVec<[BufferPoolTier; 4]>>,
119    /// Arena allocator for temporary objects
120    arena: Mutex<Bump>,
121    /// Pool statistics for monitoring
122    stats: CachePadded<PoolStatistics>,
123    /// Next buffer ID for tracking
124    next_buffer_id: AtomicUsize,
125    /// Pool configuration
126    config: PoolConfig,
127}

129#[derive(Debug)]
130#[allow(dead_code)]
131struct BufferPoolTier {
132    /// Buffers in this tier
133    buffers: Vec<Arc<ZeroCopyVideoBuffer>>,
134    /// Available buffer indices
135    available: Vec<usize>,
136    /// Buffer size for this tier
137    buffer_size: usize,
138    /// Tier ID for identification
139    tier_id: u32,
140    /// Allocation count for this tier
141    allocation_count: AtomicUsize,
142}

144#[derive(Debug, Clone)]
145#[allow(dead_code)]
146pub struct PoolConfig {
147    /// Number of buffers per tier
148    pub buffers_per_tier: usize,
149    /// Buffer sizes for different tiers (720p, 1080p, 4K, etc.)
150    pub tier_sizes: SmallVec<[usize; 4]>,
151    /// Enable pressure-adaptive allocation
152    pub adaptive_allocation: bool,
153    /// Maximum total memory usage
154    pub max_memory_bytes: usize,
155}
156
157impl Default for PoolConfig {
158    fn default() -> Self {
159        Self {
160            buffers_per_tier: 16,
161            tier_sizes: smallvec![
162                1280 * 720 * 4,  // 720p RGBA
163                1920 * 1080 * 4, // 1080p RGBA
164                3840 * 2160 * 4, // 4K RGBA
165                7680 * 4320 * 4, // 8K RGBA (future)
166            ],
167            adaptive_allocation: true,
168            max_memory_bytes: 512 * 1024 * 1024, // 512MB max
169        }
170    }

183#[allow(dead_code)]
184impl VideoBufferPool {
185    /// Create a new video buffer pool with specified configuration
186    pub fn new(config: PoolConfig) -> Result<Self, Box<dyn std::error::Error>> {
187        let mut pools = SmallVec::new();
188
189        // Pre-allocate buffer tiers
190        for (tier_id, &buffer_size) in config.tier_sizes.iter().enumerate() {
191            let mut buffers = Vec::with_capacity(config.buffers_per_tier);
192            let mut available = Vec::with_capacity(config.buffers_per_tier);
193
194            // Pre-allocate buffers for this tier
195            for i in 0..config.buffers_per_tier {
196                let buffer_id = (tier_id as u64) << 32 | (i as u64);
197                let buffer = Arc::new(ZeroCopyVideoBuffer::new(
198                    buffer_size,
199                    buffer_id,
200                    tier_id as u32,
201                )?);
202                buffers.push(buffer);
203                available.push(i);
204            }
205
206            pools.push(BufferPoolTier {
207                buffers,
208                available,
209                buffer_size,
210                tier_id: tier_id as u32,
211                allocation_count: AtomicUsize::new(0),
212            });
213        }
214
215        Ok(Self {
216            pools: RwLock::new(pools),
217            arena: Mutex::new(Bump::new()),
218            stats: CachePadded::new(PoolStatistics::default()),
219            next_buffer_id: AtomicUsize::new(1000),
220            config,
221        })
222    }
223
224    /// Acquire a buffer of specified minimum size (zero-copy)
225    pub fn acquire_buffer(&self, min_size: usize) -> Result<Arc<ZeroCopyVideoBuffer>, PoolError> {
226        self.stats.total_allocations.fetch_add(1, Ordering::Relaxed);
227
228        let pools = self.pools.read();
229
230        // Find the smallest tier that can accommodate the request
231        for tier in pools.iter() {
232            if tier.buffer_size >= min_size && !tier.available.is_empty() {
233                return self.acquire_from_tier(tier);
234            }
235        }
236
237        // No suitable buffer found in pools
238        self.stats.pool_misses.fetch_add(1, Ordering::Relaxed);
239
240        if self.config.adaptive_allocation {
241            // Try to allocate a new buffer dynamically
242            self.allocate_dynamic_buffer(min_size)
243        } else {
244            Err(PoolError::NoAvailableBuffers)
245        }
246    }
247
248    /// Acquire buffer from a specific tier
249    fn acquire_from_tier(
250        &self,
251        tier: &BufferPoolTier,
252    ) -> Result<Arc<ZeroCopyVideoBuffer>, PoolError> {
253        // This is a simplified implementation - in production, this would need
254        // proper synchronization for the available indices
255        if let Some(&index) = tier.available.last() {
256            let buffer = tier.buffers[index].clone();
257            tier.allocation_count.fetch_add(1, Ordering::Relaxed);
258            self.stats.pool_hits.fetch_add(1, Ordering::Relaxed);
259            self.stats.current_usage.fetch_add(1, Ordering::Relaxed);
260
261            // Update peak usage tracking
262            let current = self.stats.current_usage.load(Ordering::Relaxed);
263            let peak = self.stats.peak_usage.load(Ordering::Relaxed);
264            if current > peak {
265                self.stats.peak_usage.store(current, Ordering::Relaxed);
266            }
267
268            Ok(buffer)
269        } else {
270            Err(PoolError::TierExhausted)
271        }
272    }
273
274    /// Allocate a dynamic buffer when pools are exhausted
275    fn allocate_dynamic_buffer(&self, size: usize) -> Result<Arc<ZeroCopyVideoBuffer>, PoolError> {
276        let buffer_id = self.next_buffer_id.fetch_add(1, Ordering::Relaxed) as u64;
277
278        match ZeroCopyVideoBuffer::new(size, buffer_id, u32::MAX) {
279            Ok(buffer) => {
280                self.stats.current_usage.fetch_add(1, Ordering::Relaxed);
281                Ok(Arc::new(buffer))
282            }
283            Err(_) => {
284                self.stats
285                    .allocation_failures
286                    .fetch_add(1, Ordering::Relaxed);
287                Err(PoolError::AllocationFailed)
288            }
289        }
290    }
291
292    /// Release a buffer back to the pool
293    pub fn release_buffer(&self, buffer: Arc<ZeroCopyVideoBuffer>) {
294        if buffer.release() {
295            self.stats.current_usage.fetch_sub(1, Ordering::Relaxed);
296
297            // If this is a pooled buffer, return it to the available list
298            if buffer.pool_id != u32::MAX {
299                // In production, this would return the buffer to the appropriate tier
300                // For now, we just track the release
301            }
302        }
303    }
304
305    /// Get pool statistics for monitoring
306    pub fn get_statistics(&self) -> PoolStatistics {
307        PoolStatistics {
308            total_allocations: AtomicUsize::new(
309                self.stats.total_allocations.load(Ordering::Relaxed),
310            ),
311            pool_hits: AtomicUsize::new(self.stats.pool_hits.load(Ordering::Relaxed)),
312            pool_misses: AtomicUsize::new(self.stats.pool_misses.load(Ordering::Relaxed)),
313            peak_usage: AtomicUsize::new(self.stats.peak_usage.load(Ordering::Relaxed)),
314            current_usage: AtomicUsize::new(self.stats.current_usage.load(Ordering::Relaxed)),
315            allocation_failures: AtomicUsize::new(
316                self.stats.allocation_failures.load(Ordering::Relaxed),
317            ),
318        }
319    }
320
321    /// Calculate pool hit rate as percentage
322    pub fn hit_rate(&self) -> f64 {
323        let hits = self.stats.pool_hits.load(Ordering::Relaxed) as f64;
324        let total = self.stats.total_allocations.load(Ordering::Relaxed) as f64;
325
326        if total > 0.0 {
327            (hits / total) * 100.0
328        } else {
329            0.0
330        }
331    }
332
333    /// Reset pool statistics
334    pub fn reset_statistics(&self) {
335        self.stats.total_allocations.store(0, Ordering::Relaxed);
336        self.stats.pool_hits.store(0, Ordering::Relaxed);
337        self.stats.pool_misses.store(0, Ordering::Relaxed);
338        self.stats.allocation_failures.store(0, Ordering::Relaxed);
339        // Don't reset peak_usage and current_usage as they reflect current state
340    }
341}

343/// Errors that can occur during buffer pool operations
344#[derive(Debug, thiserror::Error)]
345#[allow(dead_code)]
346pub enum PoolError {
347    #[error("No available buffers in any tier")]
348    NoAvailableBuffers,
349
350    #[error("Requested tier is exhausted")]
351    TierExhausted,
352
353    #[error("Memory allocation failed")]
354    AllocationFailed,
355
356    #[error("Invalid buffer size: {size} bytes")]
357    InvalidSize { size: usize },
358
359    #[error("Pool capacity exceeded")]
360    CapacityExceeded,
361}
